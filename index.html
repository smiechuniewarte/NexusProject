'; 
    const ROOM_NAME = 'observable-global-room'; 
    const SERVER_PASSWORD = '123'; 
    
    let drone = null;
    let room = null;
    let myUsername = "";
    let membersArray = [];

    function changeScreen(hideId, showId) {
        document.getElementById(hideId).classList.remove('active');
        document.getElementById(showId).classList.add('active');
    }

    // Obsługa logowania
    function handleLogin() {
        const user = document.getElementById('usernameInput').value.trim();
        const pass = document.getElementById('passwordInput').value.trim();

        if (user === "" || pass === "") {
            alert("Uzupełnij oba pola!");
            return;
        }

        if (pass !== SERVER_PASSWORD) {
            alert("Błędne hasło serwera!");
            return;
        }

        myUsername = user;

        changeScreen('loginScreen', 'welcomeScreen');
        document.getElementById('welcomeText').innerText = `Witaj, ${myUsername}`;

        setTimeout(() => {
            changeScreen('welcomeScreen', 'chatPanel');
            initChat();
        }, 1800); 
    }

    document.getElementById('loginBtn').addEventListener('click', handleLogin);
    
    const onEnterLogin = (e) => {
        if (e.key === 'Enter') handleLogin();
    };
    document.getElementById('usernameInput').addEventListener('keydown', onEnterLogin);
    document.getElementById('passwordInput').addEventListener('keydown', onEnterLogin);

    function initChat() {
        drone = new Scaledrone(CHANNEL_ID, { data: { name: myUsername } });

        drone.on('open', error => {
            if (error) {
                console.error("Błąd Scaledrone:", error);
                return;
            }
            
            room = drone.subscribe(ROOM_NAME);
            
            room.on('message', message => {
                const { data, member, clientId } = message;

                let senderName = "Anonim";
                if (member && member.clientData && member.clientData.name) {
                    senderName = member.clientData.name;
                } else if (clientId === drone.clientId) {
                    senderName = myUsername;
                }

                const isMe = (clientId === drone.clientId || (member && member.id === drone.clientId));
                appendMessage(senderName, data, isMe);
            });

            room.on('members', members => {
                membersArray = members;
                updateMembersList();
            });

            room.on('member_join', member => {
                if (!membersArray.some(m => m.id === member.id)) {
                    membersArray.push(member);
                }
                updateMembersList();
            });

            room.on('member_leave', ({id}) => {
                const index = membersArray.findIndex(m => m.id === id);
                if (index !== -1) {
                    membersArray.splice(index, 1);
                    updateMembersList();
                }
            });
        });

        drone.on('error', err => {
            console.error("Problem z połączeniem sieciowym:", err);
        });
    }

    function updateMembersList() {
        const listDiv = document.getElementById('membersList');
        listDiv.innerHTML = '';

        membersArray.forEach(member => {
            const isMe = member.id === drone.clientId;
            const name = member.clientData && member.clientData.name ? member.clientData.name : 'Nieznany';
            
            const item = document.createElement('div');
            item.classList.add('member-item');
            if (isMe) item.classList.add('its-me');

            item.innerHTML = `
                

                ${escapeHTML(name)} ${isMe ? '(Ty)' : ''}
            `;
            listDiv.appendChild(item);
        });
    }

    const msgInput = document.getElementById('msgInput');
    const sendBtn = document.getElementById('sendBtn');
    const messagesDiv = document.getElementById('messages');

    msgInput.addEventListener('input', function() {
        this.style.height = 'auto';
        const newHeight = Math.min(this.scrollHeight, 120);
        this.style.height = newHeight + 'px';
    });

    // --- KLUCZOWA POPRAWKA: Prawidłowe wysyłanie wiadomości w Scaledrone ---
    function sendMessage() {
        const text = msgInput.value.trim();
        // Sprawdzamy czy tekst nie jest pusty i czy mamy aktywne połączenie drone
        if (text !== '' && drone) {
            drone.publish({
                room: ROOM_NAME,
                message: text
            });
            // Czyścimy pole natychmiast po wysłaniu
            msgInput.value = '';
            msgInput.style.height = '24px';
        }
    }

    sendBtn.addEventListener('click', sendMessage);
    
    msgInput.addEventListener('keydown', (e) => {
        if (e.key === 'Enter' && !e.shiftKey) {
            e.preventDefault();
            sendMessage();
        }
    });

    function escapeHTML(str) {
        if (!str) return "";
        if (typeof str !== 'string') str = String(str);
        return str.replace(/&/g, "&").replace(//g, ">");
    }

    function appendMessage(sender, text, isMe) {
        const wrapper = document.createElement('div');
        wrapper.classList.add('message-wrapper');
        if (isMe) wrapper.classList.add('me');

        wrapper.innerHTML = `
            
${escapeHTML(sender)}

            
${escapeHTML(text)}

        `;
        
        messagesDiv.appendChild(wrapper);
        messagesDiv.scrollTop = messagesDiv.scrollHeight;
    }
